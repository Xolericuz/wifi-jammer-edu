▸ FSPL Model (C++ with Emscripten)
// Free Space Path Loss formula // FSPL(dB) = 20·log10(d) + 20·log10(f) − 27.55 extern "C" { EMSCRIPTEN_KEEPALIVE float calcFSPL(float d_m, float f_mhz) { return 20*log10f(d_m) +20*log10f(f_mhz)-27.55f; } EMSCRIPTEN_KEEPALIVE float jamSignal(float pwr_dbm, float d_m, float f_mhz) { return pwr_dbm - calcFSPL(d_m,f_mhz); } EMSCRIPTEN_KEEPALIVE int deauthFrames(float sec, float rate_per_sec) { return (int)(sec * rate_per_sec); } }
▸ 802.11 Deauth Frame Structure
// Frame Control: 0xC0 0x00 = Deauth // Bytes: FC(2) Dur(2) Addr1(6) Addr2(6) Addr3(6) Seq(2) Reason(2) FCS(4) struct DeauthFrame { uint8_t fc[2] = {0xC0,0x00}; uint8_t duration[2]; uint8_t dst_addr[6]; uint8_t src_addr[6]; uint8_t bssid[6]; uint8_t seq_num[2]; uint8_t reason[2] = {0x07,0x00}; // Reason 7 = "Class 3 frame received from nonassociated STA" // FCS auto-calculated by hardware };

// Signal-to-Interference+Noise Ratio // SINR(dB) = P_signal − (P_interference + N) float calcSINR(float sig_dbm, float jam_dbm, float noise_dbm) { float interference = powf(10, jam_dbm/10); float noise = powf(10, noise_dbm/10); float signal = powf(10, sig_dbm/10); float sinr = signal / (interference + noise); return 10 * log10f(sinr); } // SINR < 0 dB → connection lost
