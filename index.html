▸ WASM Exports
// C++ functions exposed to JS:
  calcJam(pwr, dist, freq)   → dBm
  getRadius(pwr, freq, thr)  → meters
  calcSINR(sig, jam, noise)  → dB
  deauthFrames(sec, rate)    → count
  generateNoise(mean, std)   → float

▸ Build
# Requirements: emscripten SDK
emcc src/sim.cpp -o wasm/sim.js \
  -s WASM=1 -s MODULARIZE=1 \
  -s EXPORTED_FUNCTIONS='["_calcJam","_getRadius","_calcSINR"]' \
  -O3 --closure 1

# Local server
python3 -m http.server 8000
